28      })

35      @DisplayName("Testing Leap Year Logic (Decision Coverage)")
36      @ParameterizedTest
37      @CsvSource({
38          "1, 1, 0, true",     // Case: Year 0 is leap (Divisible by 400) [cite: 330]
39          "1, 1, 2000, true",  // Divisible by 400
40          "1, 1, 2024, true",  // Divisible by 4 but not 100
41          "1, 1, 1, false",    // Case: Year 1 is not leap [cite: 332]
42          "1, 1, 100, false",  // Divisible by 4 and 100, but not 400 (Not Leap)
43          "1, 1, 2023, false"  // Not divisible by 4
44      })
45      void testIsLeapYear(int day, int month, int year, boolean expectedResult) {
46          CustomDate date = new CustomDate(day, month, year);
47          assertEquals(expectedResult, date.isLeapYear(),

49      }
